## 裴波那契数列

```javascript
function fibonacci(n) {
  if (n === 0) return 1;
  if (n === 1) return 1;
  return fibonacci(n - 1) + fibonacci(n - 2);
}
```

1. 迭代法

```javascript
function fibonacciIterative(n) {
  let a = 0;
  let b = 1;
  for (let i = 2; i <= n; i++) {
    const temp = a + b;
    a = b;
    b = temp;
  }
  return b;
}
```

1. 尾递归

```javascript
function fibonacciTailRecursive(n, a = 0, b = 1) {
  if (n === 0) return a;
  if (n === 1) return b;
  return fibonacciTailRecursive(n - 1, b, a + b);
}
```

1. 动态规划

```javascript
function fibonacciDynamic(n) {
  const fib = [0, 1];
  for (let i = 2; i <= n; i++) {
    fib[i] = fib[i - 1] + fib[i - 2];
  }
  return fib[n];
}
```
